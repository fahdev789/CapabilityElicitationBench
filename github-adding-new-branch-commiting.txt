# Saving the Secret Keys at the Secret Colab environment

# It's recommended to store your GitHub PAT in Colab secrets.

from google.colab import userdata

# Retrieve the PAT from Colab secrets
GH_TOKEN = userdata.get('GITHUB_TOKEN')

if not GH_TOKEN:
    print("Error: GitHub PAT not found in Colab secrets. Please add it as 'GH_TOKEN'.")
else:
    # Construct the remote URL with the PAT
    remote_url = f"https://oauth2:{GH_TOKEN}@github.com/fahdev789/CapabilityElicitationBench.git"

    # Set the remote URL for 'origin'
    # !git remote set-url origin {remote_url}
    !git remote set-url origin {remote_url}
    print("Git remote URL updated with PAT from Colab secrets.")
